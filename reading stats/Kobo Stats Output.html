# Import all the necessary libraries for data analysis and visualization.
import json
from datetime import datetime, timedelta
import pandas as pd
import matplotlib.pyplot as plt
import calmap

import io
import base64

# Import Panel for interactive dashboarding

pn.extension() # Initialize Panel extension for Jupyter

# Suppress pandas warnings for cleaner output
warnings.filterwarnings('ignore', category=UserWarning, module='calmap')

# Load the JSON data from the file.
json_file_path = '/Users/jordanchong/my-website/reading stats/tc_readstat.json'
data = None # Initialize data to None
try:
    with open(json_file_path, 'r') as f:
        data = json.load(f)

# --- Helper Functions ---
def get_title(book_id):
    """Fetches a book title from its ID."""
    if data:
        return data['contents'].get(book_id, {}).get('title', 'Unknown Title')
    return 'Unknown Title'

def get_author(book_id):
    """Fetches a book's author from its ID."""
    if data:
        return data['contents'].get(book_id, {}).get('author', 'Unknown Author')
    return 'Unknown Author'

def get_words(book_id):
    """Fetches a book's word count from its ID."""
    if data:
        return data['contents'].get(book_id, {}).get('words', 0)
    return 0

def is_article(book_id):
    """Checks if the content is an article (book == False)."""
    if data:
        return not data['contents'].get(book_id, {}).get('book', True)
    return False

def format_duration(seconds):
    """Formats seconds into a human-readable string (e.g., '1 hours 23 minutes 45 seconds')."""
    td = timedelta(seconds=int(seconds))
    hours, remainder = divmod(td.seconds, 3600)
    minutes, seconds = divmod(remainder, 60)
    return f"{hours} hours {minutes} minutes {seconds} seconds"

def format_duration_short(seconds):
    """Formats seconds into a short h m s string (e.g., '1h23m45s')."""
    td = timedelta(seconds=int(seconds))
    hours, remainder = divmod(td.seconds, 3600)
    minutes, seconds = divmod(remainder, 60)
    return f"{hours}h{minutes}m{seconds}s"

def estimate_reading_time(word_count, wpm=200):
    """Estimates reading time in a human-readable format (e.g., '4h 23min')."""
    if wpm <= 0 or word_count <= 0:
        return "N/A"
    
    total_minutes = word_count / wpm
    hours = int(total_minutes / 60)
    minutes = int(total_minutes % 60)
    
    if hours > 0 and minutes > 0:
        return f"{hours}h {minutes}min"
    elif hours > 0:
        return f"{hours}h"
    elif minutes > 0:
        return f"{minutes}min"
    else:
        return "< 1min"


# ==============================================================================
# Cell 2: Data Preprocessing
# (This cell runs all calculations and stores results in processed_data)
# ==============================================================================
processed_data = {}
if data:
    # Reading Speed (WPM) for Finished Books
    finished_books_ids = {
        book_id for book_id, events in data['events'].items()
        if any(e['event'] == 'Finish' for e in events)
    }
    processed_data['finished_books_ids'] = finished_books_ids

    wpm_results = {}
    for book_id in finished_books_ids:
        total_duration = sum(e['duration'] for e in data['events'].get(book_id, []) if e['event'] == 'Read')
        word_count = get_words(book_id)
        if total_duration > 60 and word_count > 100:
            wpm = (word_count / total_duration) * 60
            wpm_results[get_title(book_id)] = f"{wpm:.2f} WPM"
        else:
            wpm_results[get_title(book_id)] = "WPM not available"
    processed_data['wpm_results'] = wpm_results

    # Session Data (for Longest/Average Session)
    read_durations = [
        event['duration'] for events in data['events'].values()
        for event in events if event['event'] == 'Read'
    ]
    processed_data['read_durations'] = read_durations
    
    processed_data['avg_duration_sec'] = sum(read_durations) / len(read_durations) if read_durations else 0
    processed_data['max_duration_sec'] = max(read_durations) if read_durations else 0

    # Daily Reading Streak
    read_dates = {
        datetime.fromisoformat(event['time'].replace('.000', '')).date()
        for events in data['events'].values()
        for event in events if event['event'] == 'Read'
    }
    processed_data['read_dates'] = sorted(list(read_dates))

    # Daily Reading Time for Heatmap
    daily_read_time = {}
    for events in data['events'].values():
        for event in events:
            if event['event'] == 'Read':
                read_time = datetime.fromisoformat(event['time'].replace('.000', ''))
                date = read_time.date()
                daily_read_time[date] = daily_read_time.get(date, 0) + (event['duration'] / 60) # minutes
    processed_data['daily_read_time'] = daily_read_time

    sorted_dates = processed_data['read_dates']
    longest_streak = 0
    current_streak = 0
    if sorted_dates:
        longest_streak = 1
        current_streak = 1
        for i in range(1, len(sorted_dates)):
            if sorted_dates[i] == sorted_dates[i-1] + timedelta(days=1):
                current_streak += 1
            else:
                longest_streak = max(longest_streak, current_streak)
                current_streak = 1
        longest_streak = max(longest_streak, current_streak)
        
        today = datetime.now().date()
        if sorted_dates and (today - sorted_dates[-1]).days > 1:
            current_streak = 0
        elif sorted_dates and len(sorted_dates) == 1 and (today - sorted_dates[-1]).days == 0:
             current_streak = 1
    
    # For heatmap legend
    if daily_read_time:
        min_daily_read = min(daily_read_time.values())
        max_daily_read = max(daily_read_time.values())
        processed_data['min_daily_read_time_min'] = min_daily_read
        processed_data['max_daily_read_time_min'] = max_daily_read
    else:
        processed_data['min_daily_read_time_min'] = 0
        processed_data['max_daily_read_time_min'] = 0

    # Time of Day Analysis (Total Duration)
    time_of_day_total_durations = defaultdict(int)
    for events in data['events'].values():
        for event in events:
            if event['event'] == 'Read':
                hour = datetime.fromisoformat(event['time'].replace('.000', '')).hour
                duration = event['duration']
                if 5 <= hour < 12: period = 'Morning (5am-12pm)'
                elif 12 <= hour < 17: period = 'Afternoon (12pm-5pm)'
                elif 17 <= hour < 21: period = 'Evening (5pm-9pm)'
                else: period = 'Night (9pm-5am)'
                time_of_day_total_durations[period] += duration
    processed_data['time_of_day_total_durations'] = time_of_day_total_durations

    # Vocabulary Builder
    processed_data['word_list'] = data.get('word_list', [])

    # Highlight & Note Analysis
    highlights_raw = data.get('bookmark', {})
    all_highlights_formatted = []
    for book_id, book_highlights in highlights_raw.items():
        title = get_title(book_id)
        author = get_author(book_id)
        for highlight in book_highlights:
            if 'text' in highlight:
                all_highlights_formatted.append({
                    'book_id': book_id,
                    'title': title,
                    'author': author,
                    'text': highlight['text'],
                    'time': datetime.fromisoformat(highlight['created'].replace('.000', '')) # For sorting
                })
    processed_data['all_highlights_formatted'] = sorted(all_highlights_formatted, key=lambda x: (x['book_id'], x['time']))

    # Books Currently Being Read (formerly DNF)
    books_with_reads = {
        book_id for book_id, events in data['events'].items()
        if any(e['event'] == 'Read' for e in events)
    }
    processed_data['books_currently_being_read'] = books_with_reads - finished_books_ids

    # Detailed Yearly and Monthly Breakdown (raw stats)
    book_stats_raw = defaultdict(lambda: {
        'read_sessions': [],
        'finish_date': None,
        'is_article': False
    })
    for book_id, events in data['events'].items():
        for event in events:
            dt = datetime.fromisoformat(event['time'].replace('.000', ''))
            if event['event'] == 'Read':
                book_stats_raw[book_id]['read_sessions'].append({'date': dt, 'duration': event['duration']})
            elif event['event'] == 'Finish':
                book_stats_raw[book_id]['finish_date'] = dt
            book_stats_raw[book_id]['is_article'] = is_article(book_id)
    for book_id, stats in book_stats_raw.items():
        if stats['read_sessions']:
            stats['start_date'] = min(s['date'] for s in stats['read_sessions'])
            stats['total_duration'] = sum(s['duration'] for s in stats['read_sessions'])
            stats['session_count'] = len(stats['read_sessions'])
    processed_data['book_stats_raw'] = book_stats_raw
else:
    processed_data = None # Ensure processed_data is None if data loading failed

        'finished_articles': set(), 'started_books': set(), 'continued_books': set()

             if session['date'].year == year_to_process and session['date'].month <= current_month:

                else:
                    yearly_total_book_time += session['duration']

        if stats['finish_date'] and stats['finish_date'].year == year_to_process and stats['finish_date'].month <= current_month:
            if stats['is_article']:
                yearly_finished_articles += 1
            else:
                yearly_finished_books += 1
        
        for session in stats['read_sessions']:
            if session['date'].year == year_to_process and session['date'].month <= current_month:
                month = session['date'].month
                if stats['is_article']:
                    monthly_data[month]['article_time'] += session['duration']

            monthly_data[start_month]['started_books'].add(book_id)

            else:

        f"Finished articles: {yearly_finished_articles}\n"
        f"Time reading books: {format_duration(yearly_total_book_time)} (hours: {yearly_total_book_time/3600:.2f})\n"
        f"Time reading articles: {format_duration(yearly_total_article_time)} (hours: {yearly_total_article_time/3600:.2f})\n"
        f"Total time reading: {format_duration(yearly_total_book_time + yearly_total_article_time)} (hours: {(yearly_total_book_time + yearly_total_article_time)/3600:.2f})"
    ))
    yearly_monthly_report_pane.append(pn.pane.Markdown("---"))

    yearly_monthly_report_pane.append(pn.pane.Markdown(
        f"### Daily Reading Streaks\n"
        f"🔥 Longest Reading Streak: {longest_streak} days\n"
        f"🚀 Current Reading Streak: {current_streak} days"
    ))
    yearly_monthly_report_pane.append(pn.pane.Markdown("---"))


    for month in range(1, current_month + 1):
        month_name = datetime(year_to_process, month, 1).strftime('%B')
        yearly_monthly_report_pane.append(pn.pane.Markdown(
            f"#### {month_name} {year_to_process} - Finished books: {len(monthly_data[month]['finished_books'])}, articles: {len(monthly_data[month]['finished_articles'])}, "

            f"articles: {format_duration(monthly_data[month]['article_time'])} (hours: {monthly_data[month]['article_time']/3600:.2f})"
        ))
        
        # Finished Books
        if monthly_data[month]['finished_books']:
            yearly_monthly_report_pane.append(pn.pane.Markdown("**Finished Books:**"))
            finished_books_list_items = []
            sorted_finished_books = sorted(list(monthly_data[month]['finished_books']), 
                                           key=lambda b: processed_data['book_stats_raw'][b]['finish_date'] or processed_data['book_stats_raw'][b]['start_date'] or get_title(b))
            for book_id in sorted_finished_books:
                book_info = processed_data['book_stats_raw'][book_id]
                num_days = ""
                if book_info['start_date'] and book_info['finish_date']:
                    delta = book_info['finish_date'] - book_info['start_date']
                    num_days = f" No. of Days: {delta.days + 1}"
                words = get_words(book_id)
                estimated_time = estimate_reading_time(words)
                finished_books_list_items.append(
                    f"- {get_title(book_id)} ({words} words, est. {estimated_time}) - {get_author(book_id)} "
                    f"(Duration: {format_duration_short(book_info['total_duration'])} over {book_info['session_count']} Sessions) "

                )
            yearly_monthly_report_pane.append(pn.pane.Markdown("\n".join(finished_books_list_items)))
        
        # Started Books (not finished this month)
        started_not_finished = monthly_data[month]['started_books'] - monthly_data[month]['finished_books']
        if started_not_finished:

            started_list_items = []
            sorted_started_not_finished = sorted(list(started_not_finished), 
                                                 key=lambda b: processed_data['book_stats_raw'][b]['start_date'] or get_title(b))

                sessions_this_month = sum(1 for s in book_info['read_sessions'] if s['date'].month == month and s['date'].year == year_to_process)

                estimated_time = estimate_reading_time(words)
                started_list_items.append(
                    f"- {get_title(book_id)} ({words} words, est. {estimated_time}) - {get_author(book_id)} "
                    f"(Duration: {format_duration_short(duration_this_month)} over {sessions_this_month} Sessions) "
                    f"Started: {book_info['start_date'].strftime('%Y-%m-%d')}"
                )
            yearly_monthly_report_pane.append(pn.pane.Markdown("\n".join(started_list_items)))
                  
        # Continued Books
        if monthly_data[month]['continued_books']:
            yearly_monthly_report_pane.append(pn.pane.Markdown("**Continued Books:**"))
            continued_list_items = []
            sorted_continued_books = sorted(list(monthly_data[month]['continued_books']), 
                                            key=lambda b: processed_data['book_stats_raw'][b]['start_date'] or get_title(b))
            for book_id in sorted_continued_books:
                book_info = processed_data['book_stats_raw'][book_id]
                duration_this_month = sum(s['duration'] for s in book_info['read_sessions'] if s['date'].month == month and s['date'].year == year_to_process)
                sessions_this_month = sum(1 for s in book_info['read_sessions'] if s['date'].month == month and s['date'].year == year_to_process)
                words = get_words(book_id)
                estimated_time = estimate_reading_time(words)
                continued_list_items.append(
                    f"- {get_title(book_id)} ({words} words, est. {estimated_time}) - {get_author(book_id)} "
                    f"(Duration: {format_duration_short(duration_this_month)} over {sessions_this_month} Sessions) "
                    f"Started: {book_info['start_date'].strftime('%Y-%m-%d')}"
                )
            yearly_monthly_report_pane.append(pn.pane.Markdown("\n".join(continued_list_items)))
    
    yearly_monthly_report_pane.append(pn.pane.Markdown("---")) # Final separator

        ax.bar(sorted_total_durations.keys(), total_durations_hours, color=['skyblue', 'orange', 'salmon', 'darkblue'])
        ax.set_title('Total Reading Time by Time of Day', fontsize=16)
        ax.set_ylabel('Total Reading Time (Hours)')
        plt.xticks(rotation=15)
        fig.savefig(buffer, format='png', bbox_inches='tight')
        plt.close(fig)
        buffer.seek(0)
        time_of_day_plot_img_base64 = base64.b64encode(buffer.read()).decode('utf-8')
        
        # FIX: Use pn.pane.HTML to embed the Base64 image string
        time_of_day_pane.append(pn.pane.HTML(f'<img src="data:image/png;base64,{time_of_day_plot_img_base64}" style="width="450"; height="270";">', sizing_mode='scale_width'))
    else:
        time_of_day_pane.append(pn.pane.Markdown("No reading session data found for time of day analysis."))

    time_of_day_list_items = [f"- {period}: {duration_sec / 3600:.2f} hours" for period, duration_sec in sorted_total_durations.items()]
    time_of_day_pane.append(pn.pane.Markdown("\n".join(time_of_day_list_items)))


    # --- Highlights & Vocabulary Content ---
    highlights_vocab_pane.append(pn.pane.Markdown("### My Word List"))
    vocab_list = processed_data['word_list']
    if vocab_list:
        vocab_list_items = [f"- {word}" for word in vocab_list]

                current_book_id = highlight['book_id']

            words = get_words(book_id)
            estimated_time = estimate_reading_time(words)
            
            currently_reading_list_items.append(f"- {title} ({words} words, est. {estimated_time}) by {author}")
        books_being_read_pane.append(pn.pane.Markdown("\n".join(currently_reading_list_items))) # Join and add as a single Markdown pane
    else:
        books_being_read_pane.append(pn.pane.Markdown("No books currently being read. You finish everything or haven't started anything yet!"))


else: # If no data loaded at all, display an error message in each pane
    yearly_monthly_report_pane.append(pn.pane.Markdown("### Error: Data not loaded. Please check the JSON file path and content."))
    streaks_calendar_pane.append(pn.pane.Markdown("### Error: Data not loaded. Please check the JSON file path and content."))
